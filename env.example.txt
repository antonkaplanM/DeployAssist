# Atlassian API Configuration
# Copy this file to .env and fill in your actual values
# DO NOT commit .env file to version control

# Your Atlassian email address
ATLASSIAN_EMAIL=your.email@company.com

# Your Atlassian API token (get from https://id.atlassian.com/manage-profile/security/api-tokens)
ATLASSIAN_API_TOKEN=your_api_token_here

# Your Atlassian Cloud ID (UUID format)
ATLASSIAN_CLOUD_ID=your-cloud-id-here

# Atlassian API base URL (usually doesn't need to change)
ATLASSIAN_BASE_URL=https://api.atlassian.com/ex/jira

# Application Configuration
PORT=8080
NODE_ENV=development
